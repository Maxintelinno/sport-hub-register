booking_items
เก็บรายละเอียดช่วงเวลาที่จองจริง
CREATE TABLE booking_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    booking_id UUID NOT NULL REFERENCES bookings(id) ON DELETE CASCADE,
    field_id UUID NOT NULL REFERENCES fields(id),
    court_id UUID NOT NULL REFERENCES field_courts(id),
    booking_date DATE NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    start_at TIMESTAMP NOT NULL,
    end_at TIMESTAMP NOT NULL,
    price_per_hour NUMERIC(10,2) NOT NULL,
    total_amount NUMERIC(10,2) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'confirmed',
    created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

ทำไมต้องมีทั้ง start_time/end_time และ start_at/end_at เพราะใช้งานคนละแบบ
- start_time, end_time เอาไว้แสดงผล/ค้นหาง่าย
- start_at, end_at เอาไว้ตรวจ overlap แบบจริงจัง
เช่น
- booking_date = 2026-03-12
- start_time = 18:00
- end_time = 20:00

จะกลายเป็น
- start_at = 2026-03-12 18:00:00
= end_at = 2026-03-12 20:00:00

จุดสำคัญสุด: กันจองซ้ำ

PostgreSQL ทำได้สวยมากด้วย exclusion constraint
ก่อนอื่นเปิด extension:

CREATE EXTENSION IF NOT EXISTS btree_gist;

แล้วสร้าง constraint:

ALTER TABLE booking_items
ADD CONSTRAINT no_overlapping_booking
EXCLUDE USING gist (
    court_id WITH =,
    tsrange(start_at, end_at) WITH &&
)
WHERE (status IN ('confirmed', 'pending'));

2) payment_webhook_logs

เอาไว้เก็บ “ของดิบ” ที่ provider ส่งมา
เอาไว้ตอบคำถามว่า:
- provider ส่งอะไรเข้ามา
- header/signature เป็นอะไร
- เราได้รับเมื่อไหร่
- ตอน parse ล้มเหลวไหม
- debug ปัญหาย้อนหลังได้ไหม

ตารางนี้สำคัญมากเวลาเกิดปัญหาแบบ:
- เงินเข้า แต่ booking ไม่ confirmed
- webhook ยิงมาแปลก
- signature verify ไม่ผ่าน
- provider ยิง event format ใหม่มา

ดังนั้น ควรมี
ตาราง payment_webhook_logs
เก็บ webhook ดิบไว้ audit และ debug
CREATE TABLE payment_webhook_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    provider VARCHAR(50) NOT NULL,
    event_type VARCHAR(100) NULL,
    event_id VARCHAR(150) NULL,
    signature TEXT NULL,
    request_headers JSONB NULL,
    request_body JSONB NOT NULL,
    received_at TIMESTAMP NOT NULL DEFAULT NOW(),
    processed_at TIMESTAMP NULL,
    process_status VARCHAR(30) NOT NULL DEFAULT 'received',
    process_error TEXT NULL
);

process_status
- received
- processed
- ignored
- failed

ตาราง payment_events
เก็บ event ที่ตีความแล้ว ป้องกัน webhook ซ้ำ
CREATE TABLE payment_events (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    provider VARCHAR(50) NOT NULL,
    event_id VARCHAR(150) NOT NULL,
    payment_id UUID NULL REFERENCES payments(id) ON DELETE SET NULL,
    booking_id UUID NULL REFERENCES bookings(id) ON DELETE SET NULL,
    event_type VARCHAR(100) NOT NULL,
    event_status VARCHAR(50) NULL,
    payload JSONB NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    UNIQUE(provider, event_id)
);
อันนี้สำคัญมาก เพราะ webhook มักยิงซ้ำได้

3) payment_events
เอาไว้เก็บ event ที่ “ระบบยอมรับและประมวลผลแล้ว”
จุดประสงค์หลักคือ:
- กัน webhook ซ้ำ

เพราะ provider มักส่ง webhook event เดิมซ้ำได้

เช่น event_id เดิมมา 2–3 รอบ
ถ้าไม่มีตารางนี้ คุณอาจ:
- update payment ซ้ำ
- ยิง notification ซ้ำ
- เปลี่ยน booking state ซ้ำ
ดังนั้น UNIQUE(provider, event_id) สำคัญมาก

CREATE TABLE payments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    booking_id UUID NOT NULL REFERENCES bookings(id) ON DELETE CASCADE,

    payment_no VARCHAR(30) NOT NULL UNIQUE,

    provider VARCHAR(50) NOT NULL, -- promptpay, omise, etc
    method VARCHAR(30) NOT NULL,   -- qr, card

    amount NUMERIC(10,2) NOT NULL,
    currency VARCHAR(10) NOT NULL DEFAULT 'THB',

    status VARCHAR(20) NOT NULL DEFAULT 'pending',
    -- pending, processing, paid, failed, expired

    -- provider mapping
    provider_payment_id VARCHAR(100) NULL,
    provider_transaction_id VARCHAR(150) NULL,
    provider_reference VARCHAR(150) NULL,

    -- QR
    qr_payload TEXT NULL,
    qr_image_url TEXT NULL,

    -- lifecycle
    expires_at TIMESTAMP NOT NULL,
    paid_at TIMESTAMP NULL,
    failed_at TIMESTAMP NULL,
    failure_reason TEXT NULL,

    metadata JSONB NULL,

    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP NOT NULL DEFAULT NOW()
);

🔥 Mapping ระหว่าง bookings กับ payments
bookings.id = payments.booking_id

🔄 Flow ที่จะใช้ field ใหม่
ตอน create booking

status = pending
payment_status = unpaid
expires_at = now + 10 min

ตอนจ่ายเงินสำเร็จ
booking.status = confirmed
booking.payment_status = paid
booking.paid_at = now()

ตอนหมดเวลา
booking.status = expired
booking.payment_status = failed
booking.cancelled_at = now()
cancel_reason = 'timeout'
